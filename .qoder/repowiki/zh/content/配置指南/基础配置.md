# 基础配置指南

## 配置文件说明

Kewen Framework 支持多种配置方式，推荐使用 YAML 格式。

## 基础配置

### 服务器配置

```yaml
server:
  port: 8081                    # 服务端口
  servlet:
    context-path: /api          # 上下文路径（可选）
  tomcat:
    uri-encoding: UTF-8         # URI 编码
    threads:
      max: 200                  # 最大线程数
      min-spare: 10             # 最小空闲线程数
```

### 应用配置

```yaml
spring:
  application:
    name: kewen-framework-app   # 应用名称
  profiles:
    active: dev                 # 激活的环境
```

## 数据源配置

### MySQL 配置

```yaml
spring:
  datasource:
    url: jdbc:mysql://localhost:3306/kewen_framework?useUnicode=true&characterEncoding=utf8&useSSL=false&serverTimezone=Asia/Shanghai
    username: root
    password: your_password
    driver-class-name: com.mysql.cj.jdbc.Driver
    
    # HikariCP 连接池配置
    hikari:
      auto-commit: true
      connection-test-query: SELECT 1
      connection-timeout: 30000         # 连接超时 30秒
      idle-timeout: 600000              # 空闲超时 10分钟
      max-lifetime: 1800000             # 最大生命周期 30分钟
      maximum-pool-size: 20             # 最大连接数
      minimum-idle: 5                   # 最小空闲连接
      validation-timeout: 5000          # 验证超时 5秒
      pool-name: HikariPool-Kewen       # 连接池名称
```

### P6Spy SQL 监控（开发环境）

```yaml
spring:
  datasource:
    driver-class-name: com.kewen.framework.boot.basic.p6spy.KP6SpyDriver
```

## MyBatis-Plus 配置

```yaml
mybatis-plus:
  mapper-locations: classpath*:/mapper/**/*.xml
  type-aliases-package: com.kewen.framework.**.entity
  configuration:
    map-underscore-to-camel-case: true
    cache-enabled: false
    log-impl: org.apache.ibatis.logging.stdout.StdOutImpl  # 开发环境打印 SQL
  global-config:
    db-config:
      id-type: auto                 # 主键策略
      logic-delete-field: deleted   # 逻辑删除字段
      logic-delete-value: 1
      logic-not-delete-value: 0
```

## 权限系统配置

### 安全配置

```yaml
kewen:
  security:
    # 登录配置
    login:
      login-url: /login                     # 登录接口地址
      current-user-url: /currentUser        # 当前用户接口
      username-parameter: username          # 用户名参数
      password-parameter: password          # 密码参数
      token-parameter: Authorization        # Token 请求头参数
    
    # Session 配置
    session:
      maximum-sessions: 10                  # 最大 Session 数
      max-sessions-prevents-login: false    # 是否阻止重复登录（false=允许踢人）
    
    # Remember Me 配置
    remember-me:
      enabled: true
      remember-parameter: remember-me
      validity-seconds: 2592000             # 有效期 30天
```

### 权限数据表配置

```yaml
kewen:
  auth:
    auth-data-table:
      table-name: sys_auth_data             # 权限表名
      id-column: id                         # 主键列
      data-id-column: data_id               # 数据 ID 列
      authority-column: authority           # 权限列
      description-column: description       # 描述列
      business-function-column: business_function  # 业务功能列
      operate-column: operate               # 操作列
    
    cache-auth: false                       # 是否缓存菜单权限
```

## 请求日志配置

```yaml
kewen:
  request:
    # 方糖消息通知
    message:
      fang-tang: true                       # 是否发送方糖消息
    
    # 请求持久化
    persistent:
      database: true                        # 是否持久化到数据库

# 方糖配置
kewen:
  message:
    fang-tang:
      key: your_sct_key                     # 方糖 API Key
      domain: https://sctapi.ftqq.com       # 方糖 API 域名
```

## 存储模块配置

### 七牛云存储

```yaml
kewen:
  storage:
    platform: qiniu                         # 存储平台
    qiniu:
      access-key: your_access_key
      secret-key: your_secret_key
      bucket: your_bucket_name
      domain: https://your-domain.com       # 域名
      region: huanan                        # 区域（huanan=华南）
```

### 本地存储

```yaml
kewen:
  storage:
    platform: local
    local:
      base-path: /data/files                # 基础路径
      domain: http://localhost:8081/files   # 访问域名
```

## 多租户配置

```yaml
kewen:
  tenant:
    open: true                              # 是否开启多租户
    column: tenant_id                       # 租户字段名
    ignore-tables:                          # 忽略多租户的表
      - sys_config
      - sys_dict
```

## 日志配置

### logback-spring.xml

```xml
<configuration>
    <springProperty scope="context" name="app_name" source="spring.application.name"/>
    
    <include resource="org/springframework/boot/logging/logback/defaults.xml"/>
    <include resource="org/springframework/boot/logging/logback/console-appender.xml"/>
    
    <springProperty name="logging.path" source="logging.file.path" defaultValue="./logs"/>
    <springProperty name="logging.name" source="logging.file.name" defaultValue="app"/>
    
    <!-- 文件日志 -->
    <appender name="rollingFile" class="ch.qos.logback.core.rolling.RollingFileAppender">
        <file>${logging.path}/${logging.name}.log</file>
        <rollingPolicy class="ch.qos.logback.core.rolling.SizeAndTimeBasedRollingPolicy">
            <fileNamePattern>${logging.path}/${logging.name}-%d{yyyy-MM-dd}.%i.log</fileNamePattern>
            <maxHistory>30</maxHistory>
            <maxFileSize>10MB</maxFileSize>
            <totalSizeCap>2GB</totalSizeCap>
        </rollingPolicy>
        <encoder>
            <pattern>%d{yyyy-MM-dd HH:mm:ss.SSS} [%thread] %-5level %logger{36} - %msg%n</pattern>
        </encoder>
    </appender>
    
    <!-- 日志级别 -->
    <root level="INFO">
        <appender-ref ref="console"/>
        <appender-ref ref="rollingFile"/>
    </root>
    
    <!-- 特定包日志级别 -->
    <logger name="com.kewen.framework" level="DEBUG"/>
    <logger name="org.springframework.security" level="DEBUG"/>
</configuration>
```

### application.yml 日志配置

```yaml
logging:
  level:
    root: INFO
    com.kewen.framework: DEBUG
    org.springframework.security: DEBUG     # 开发环境调试 Security
  file:
    name: app                               # 日志文件名
    path: ./logs                            # 日志目录
```

kewen:

5. application.yml
